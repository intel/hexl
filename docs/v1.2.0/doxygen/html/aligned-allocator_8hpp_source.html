63  if (!IsPowerOfTwo(Alignment)) {
64  return nullptr;
65  }
66  // Allocate enough space to ensure the alignment can be satisfied
67  size_t buffer_size = sizeof(T) * n + Alignment;
68  // Additionally, allocate a prefix to store the memory location of the
69  // unaligned buffer
70  size_t alloc_size = buffer_size + sizeof(void*);
71  void* buffer = m_alloc_impl->allocate(alloc_size);
72  if (!buffer) {
73  return nullptr;
74  }
75

77  void* aligned_buffer = static_cast<char*>(buffer) + sizeof(void*);
78  std::align(Alignment, sizeof(T) * n, aligned_buffer, buffer_size);
79  if (!aligned_buffer) {
80  return nullptr;
81  }

84  void* store_buffer_addr =
85  static_cast<char*>(aligned_buffer) - sizeof(void*);
86  *(static_cast<void**>(store_buffer_addr)) = buffer;
