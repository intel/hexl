60 inline void LeftShift128(uint64_t* result_hi, uint64_t* result_lo,
61  const uint64_t op_hi, const uint64_t op_lo,
62  const uint64_t shift_value) {
63  HEXL_CHECK(result_hi != nullptr, "Require result_hi != nullptr");
64  HEXL_CHECK(result_lo != nullptr, "Require result_lo != nullptr");
65  HEXL_CHECK(shift_value <= 128,
66  "shift_value cannot be greater than 128 " << shift_value);
67 
68  if (shift_value == 0) {
69  *result_hi = op_hi;
70  *result_lo = op_lo;
71  } else if (shift_value == 64) {
72  *result_hi = op_lo;
73  *result_lo = 0ULL;
74  } else if (shift_value == 128) {
75  *result_hi = 0ULL;
76  *result_lo = 0ULL;
77  } else if (shift_value >= 1 && shift_value <= 63) {
78  *result_hi = (op_hi << shift_value) | (op_lo >> (64 - shift_value));
79  *result_lo = op_lo << shift_value;
80  } else if (shift_value >= 65 && shift_value < 128) {
81  *result_hi = op_lo << (shift_value - 64);
82  *result_lo = 0ULL;
83  }
84 }

155 inline uint64_t SignificantBitLength(const uint64_t* value) {
156  HEXL_CHECK(value != nullptr, "Require value != nullptr");
157 
158  unsigned long count = 0; // NOLINT(runtime/int)
159 
160  // second 64bit block
161  _BitScanReverse64(&count, *(value + 1));
162  if (count >= 0 && *(value + 1) > 0) {
163  return static_cast<uint64_t>(count) + 1 + 64;

166  // first 64bit block

188 inline void DivideUInt128UInt64(uint64_t* quotient, const uint64_t* numerator,
189  const uint64_t denominator) {
190  HEXL_CHECK(quotient != nullptr, "Require quotient != nullptr");
191  HEXL_CHECK(numerator != nullptr, "Require numerator != nullptr");
192  HEXL_CHECK(denominator != 0, "denominator cannot be 0 " << denominator);
193 
194  // get bit count of divisor
195  uint64_t numerator_bits = SignificantBitLength(numerator);
196  const uint64_t numerator_bits_const = numerator_bits;
197  const uint64_t uint_128_bit = 128ULL;
198 
199  uint64_t MASK[2]{0x0000000000000001, 0x0000000000000000};
200  uint64_t remainder[2]{0, 0};
201  uint64_t quotient_temp[2]{0, 0};
202  uint64_t denominator_temp[2]{denominator, 0};
203 
204  quotient[0] = numerator[0];
205  quotient[1] = numerator[1];
206 
207  // align numerator
208  LeftShift128(&quotient[1], &quotient[0], quotient[1], quotient[0],
209  (uint_128_bit - numerator_bits_const));
210 
211  while (numerator_bits) {
212  // if remainder is negative
213  if (CheckSign(remainder)) {
214  LeftShift128(&remainder[1], &remainder[0], remainder[1], remainder[0], 1);
215  RightShift128(&quotient_temp[1], &quotient_temp[0], quotient[1],
216  quotient[0], (uint_128_bit - 1));
217  remainder[0] = remainder[0] | quotient_temp[0];
218  LeftShift128(&quotient[1], &quotient[0], quotient[1], quotient[0], 1);
219  // remainder=remainder+denominator_temp
220  AddWithCarry128(&remainder[1], &remainder[0], remainder[1], remainder[0],
221  denominator_temp[1], denominator_temp[0]);
222  } else { // if remainder is positive
223  LeftShift128(&remainder[1], &remainder[0], remainder[1], remainder[0], 1);
224  RightShift128(&quotient_temp[1], &quotient_temp[0], quotient[1],
225  quotient[0], (uint_128_bit - 1));
226  remainder[0] = remainder[0] | quotient_temp[0];
227  LeftShift128(&quotient[1], &quotient[0], quotient[1], quotient[0], 1);
228  // remainder=remainder-denominator_temp
229  SubWithCarry128(&remainder[1], &remainder[0], remainder[1], remainder[0],
230  denominator_temp[1], denominator_temp[0]);
231  }
232 
233  // if remainder is positive set MSB of quotient[0]=1
234  if (!CheckSign(remainder)) {
235  MASK[0] = 0x0000000000000001;
236  MASK[1] = 0x0000000000000000;
237  LeftShift128(&MASK[1], &MASK[0], MASK[1], MASK[0],
238  (uint_128_bit - numerator_bits_const));
239  quotient[0] = quotient[0] | MASK[0];
240  quotient[1] = quotient[1] | MASK[1];
241  }
242  quotient_temp[0] = 0;
243  quotient_temp[1] = 0;
244  numerator_bits--;
245  }
246 
247  if (CheckSign(remainder)) {
248  // remainder=remainder+denominator_temp
249  AddWithCarry128(&remainder[1], &remainder[0], remainder[1], remainder[0],
250  denominator_temp[1], denominator_temp[0]);
251  }
252  RightShift128(&quotient[1], &quotient[0], quotient[1], quotient[0],
253  (uint_128_bit - numerator_bits_const));
254 }
